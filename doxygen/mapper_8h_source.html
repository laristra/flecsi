251  virtual void slice_task(const Legion::Mapping::MapperContext ctx,
252  const Legion::Task & task,
253  const Legion::Mapping::Mapper::SliceTaskInput & input,
254  Legion::Mapping::Mapper::SliceTaskOutput & output) {
255  using legion_proc = Legion::Processor;
256  context_t & context_ = context_t::instance();
257 
258  if(task.tag == FLECSI_MAPPER_SUBRANK_LAUNCH) {
259  // expect a 1-D index domain
260  assert(input.domain.get_dim() == 1);
261  // send the whole domain to our local processor
262  output.slices.resize(1);
263  output.slices[0].domain = input.domain;
264  output.slices[0].proc = task.target_proc;
265  return;
266  } // end if FLECSI_MAPPER_SUBRANK_LAUNCH
267 
268  if(task.tag == FLECSI_MAPPER_FORCE_RANK_MATCH) {
269  // expect a 1-D index domain - each point goes to the corresponding node
270  assert(input.domain.get_dim() == 1);
271  LegionRuntime::Arrays::Rect<1> r = input.domain.get_rect<1>();
272 
273  // go through all the CPU processors and find a representative for each
274  // node (i.e. address space)
275  std::map<int, Legion::Processor> targets;
276 
277  Legion::Machine::ProcessorQuery pq =
278  Legion::Machine::ProcessorQuery(machine).only_kind(
279  Legion::Processor::LOC_PROC);
280  for(Legion::Machine::ProcessorQuery::iterator it = pq.begin();
281  it != pq.end();
282  ++it) {
283  Legion::Processor p = *it;
284  int a = p.address_space();
285  if(targets.count(a) == 0)
286  targets[a] = p;
287  }
288 
289  output.slices.resize(1);
290  for(int a = r.lo[0]; a <= r.hi[0]; a++) {
291  assert(targets.count(a) > 0);
292  output.slices[0].domain = // Legion::Domain::from_rect<1>(
293  Legion::Rect<1>(a, a);
294  output.slices[0].proc = targets[a];
295  }
296  return;
297  } // FLECSI_MAPPER_FORCE_RANK_MATCH
298 
299  DefaultMapper::slice_task(ctx, task, input, output);
300  // end else
301  }
