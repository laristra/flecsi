190  node_vector_t sort() {
191  node_vector_t sorted;
192 
193  // Create a list of the nodes
194  node_list_t nodes;
195  for(auto n = nodes_.begin(); n != nodes_.end(); ++n) {
196  nodes.push_back(n->second);
197  } // for
198 
199  // Create a tally of the number of dependencies of each node
200  // in the graph.
201  std::queue<node_t> q;
202 
203  for(auto n = nodes.begin(); n != nodes.end();) {
204  if(n->edges().size() == 0) {
205  q.push(*n);
206  n = nodes.erase(n);
207  }
208  else {
209  ++n;
210  } // if
211  } // for
212 
213  size_t count{0};
214  while(!q.empty()) {
215  auto root = q.front();
216  sorted.push_back(root);
217  q.pop();
218 
219  for(auto n = nodes.begin(); n != nodes.end();) {
220  auto it = std::find(n->edges().begin(), n->edges().end(), root.hash());
221 
222  if(it != n->edges().end()) {
223  n->edges().erase(it);
