305  template<size_t NAMESPACE, typename OBJECT_TYPE, typename... ARGS>
306  OBJECT_TYPE * add_global_object(size_t index, ARGS &&... args) {
307  size_t KEY = NAMESPACE ^ index;
308 
309  flog_assert(
310  task_depth() == 0, "you cannot add global objects from within a task");
311 
312  flog_assert(
313  global_object_registry_.find(KEY) == global_object_registry_.end(),
314  "global key already exists");
315 
316  auto ptr = new OBJECT_TYPE(std::forward<ARGS>(args)...);
317 
318  flog(internal) << "Adding global object" << std::endl
319  << "\tindex: " << index << std::endl
320  << "\thash: " << NAMESPACE << std::endl
321  << "\ttype: " << utils::demangle(typeid(OBJECT_TYPE).name())
322  << std::endl
323  << "\taddress: " << ptr << std::endl;
324 
325  std::get<0>(global_object_registry_[KEY]) =
326  reinterpret_cast<uintptr_t>(ptr);
327 
328  using wrapper_t = global_object_wrapper_u<OBJECT_TYPE>;
329  std::get<1>(global_object_registry_[KEY]) = &wrapper_t::cleanup;
