120  static_assert(processor_type != task_processor_type_t::mpi,
121  "Legion tasks cannot use MPI");
122 
123  const std::string name = utils::symbol<*TASK>();

132  : Legion::Processor::LOC_PROC;
133  registrar.add_constraint(Legion::ProcessorConstraint(kind));
134  registrar.set_leaf(leaf_task(A));
