88  /*-------------------------------------------------------------------------*
89  * Clear the graph. This call is a little counter-intuitive. It frees
90  * the currently allocated graph and creates a new empty one in its place.
91  * Therefore, there is always a graph instance available.
92  *-------------------------------------------------------------------------*/
93 
94  void clear() {
95  if(graph_ != nullptr) {
96  gvFreeLayout(gvc_, graph_);
97  agclose(graph_);
98  } // if
99 
100  graph_ = agopen(GV_GRAPH, Agdirected, nullptr);
101 
102  // set default node attributes
103  agattr(graph_, AGNODE, GV_LABEL, GV_LABEL_DEFAULT);
104  agattr(graph_, AGNODE, GV_COLOR, GV_COLOR_BLACK);
105  agattr(graph_, AGNODE, GV_STYLE, GV_STYLE_DEFAULT);
106  agattr(graph_, AGNODE, GV_FILL_COLOR, GV_COLOR_LIGHTGREY);
107  agattr(graph_, AGNODE, GV_FONT_COLOR, GV_COLOR_BLACK);
108 
109  // set default edge attributes
110  agattr(graph_, AGEDGE, GV_DIR, GV_DIR_DEFAULT);
111  agattr(graph_, AGEDGE, GV_LABEL, GV_LABEL_DEFAULT);
112  agattr(graph_, AGEDGE, GV_COLOR, GV_COLOR_BLACK);
113  agattr(graph_, AGEDGE, GV_STYLE, GV_STYLE_DEFAULT);
114  agattr(graph_, AGEDGE, GV_FILL_COLOR, GV_COLOR_BLACK);
115  agattr(graph_, AGEDGE, GV_FONT_COLOR, GV_COLOR_BLACK);
116  agattr(graph_, AGEDGE, GV_HEADPORT, GV_HEADPORT_DEFAULT);
117  agattr(graph_, AGEDGE, GV_TAILPORT, GV_TAILPORT_DEFAULT);
118  } // clear

120  /*-------------------------------------------------------------------------*

123  * If 'label' is non-null, it will be used for the display name of the node.

125 
126  Agnode_t * add_node(const char * name, const char * label = nullptr) {
127  char buffer[1024];
128  sprintf(buffer, "%s", name);
129  Agnode_t * node = agnode(graph_, buffer, ag_create);
130 
131  if(label != nullptr) {
132  char attr[1024];
133  sprintf(attr, "%s", "label");
