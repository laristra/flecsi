433  void build_reverse_intermediate_maps(bool reset = false, bool sort = false) {
434 
435  // clear the map for safety
436  if (reset)
437  reverse_intermediate_map_.clear();
438 
439  // now flip all the mappings
440  for (const auto & forward_map : intermediate_map_) {
441  auto key = forward_map.first;
442  auto & reverse_map = reverse_intermediate_map_[key];
443  // it will be empty if never set, or reset==true
444  if (reverse_map.empty()) {
445 
446  // assume unsorted
447  if (sort) {
448  for (auto & entry : forward_map.second) {
449  std::sort(entry.second.begin(), entry.second.end());
450  reverse_map[entry.second] = entry.first;
451  }
452  }
453  // assume sorted
454  else {
455  for (const auto & entry : forward_map.second)
456  reverse_map[entry.second] = entry.first;
457  }
458  }
459  }
460  }
