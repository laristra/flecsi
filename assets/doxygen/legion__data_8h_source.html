255  void add_index_space(
256  size_t index_space_id,
257  const coloring_info_map_t & coloring_info_map,
258  const sparse_index_space_info_t* sparse_info) {
259  using namespace std;
260 
261  using namespace Legion;
262  using namespace LegionRuntime;
263  using namespace Arrays;
264 
265  using namespace execution;
266 
267  context_t & context = context_t::instance();
268 
269  index_space_t is;
270  is.index_space_id = index_space_id;
271 
272  // Create expanded IndexSpace
273  index_spaces_.insert(index_space_id);
274 
275  // Determine max size of a color partition
276  is.total_num_entities = 0;
277  for (auto color_idx : coloring_info_map) {
278  {
279  clog_tag_guard(legion_data);
280  clog(trace) << "index: " << index_space_id
281  << " color: " << color_idx.first << " " << color_idx.second
282  << std::endl;
283  } // scope
284 
285  is.total_num_entities = std::max(
286  is.total_num_entities, color_idx.second.exclusive +
287  color_idx.second.shared +
288  color_idx.second.ghost);
289  } // for color_idx
290 
291  {
292  clog_tag_guard(legion_data);
293  clog(trace) << "total_num_entities " << is.total_num_entities
294  << std::endl;
295  } // scope
296 
297  // Create expanded index space
298  LegionRuntime::Arrays::Rect<2> expanded_bounds =
299  LegionRuntime::Arrays::Rect<2>(
300  LegionRuntime::Arrays::Point<2>::ZEROES(),
301  make_point(num_colors_, is.total_num_entities));
302 
303  Domain expanded_dom(Domain::from_rect<2>(expanded_bounds));
304 
305  is.index_space = runtime_->create_index_space(ctx_, expanded_dom);
306  attach_name(is, is.index_space, "expanded index space");
307 
308  // Read user + FleCSI registered field spaces
309  is.field_space = runtime_->create_field_space(ctx_);
310 
311  attach_name(is, is.field_space, "expanded field space");
312 
313  if(sparse_info){
314  is.has_sparse_fields = true;
315 
316  sparse_index_space_t sis;
317  // TODO: need to formalize this index space offset scheme
318  sis.index_space_id = index_space_id + 8192;
319 
320  sis.max_shared_ghost = 0;
321 
322  for (auto color_idx : coloring_info_map) {
323  sis.max_shared_ghost = std::max(
324  sis.max_shared_ghost, color_idx.second.shared + color_idx.second.ghost);
325  }
326 
327  sis.exclusive_reserve = sparse_info->exclusive_reserve;
328  sis.max_entries_per_index = sparse_info->max_entries_per_index;
329 
330  sis.color_size =
331  sis.exclusive_reserve +
332  sis.max_shared_ghost * sis.max_entries_per_index;
333 
334  // Create expanded index space
335  LegionRuntime::Arrays::Rect<2> expanded_bounds =
336  LegionRuntime::Arrays::Rect<2>(
337  LegionRuntime::Arrays::Point<2>::ZEROES(),
338  make_point(num_colors_, sis.color_size));
339 
340  Domain expanded_dom(Domain::from_rect<2>(expanded_bounds));
341 
342  sis.index_space = runtime_->create_index_space(ctx_, expanded_dom);
343  attach_name(sis, sis.index_space, "expanded sparse index space");
344 
345  // Read user + FleCSI registered field spaces
346  sis.field_space = runtime_->create_field_space(ctx_);
347 
348  attach_name(sis, sis.field_space, "expanded sparse field space");
349 
350  sparse_index_space_map_[index_space_id] = std::move(sis);
351  }
352 
353  index_space_map_[index_space_id] = std::move(is);
354  }

359  void add_adjacency(const adjacency_info_t & adjacency_info) {
360  using namespace std;
361 
362  using namespace Legion;
363  using namespace LegionRuntime;
364  using namespace Arrays;
365 
366  using namespace execution;
367 
368  context_t & context = context_t::instance();
369 
370  using field_info_t = context_t::field_info_t;
371 
372  adjacency_t c;
373 
374  clog_assert(
375  adjacencies_.find(adjacency_info.index_space) == adjacencies_.end(),
376  "adjacency exists");
377 
378  adjacencies_.insert(adjacency_info.index_space);
379 
380  c.index_space_id = adjacency_info.index_space;
381  c.from_index_space_id = adjacency_info.from_index_space;
382  c.to_index_space_id = adjacency_info.to_index_space;
383 
384  auto fitr = index_space_map_.find(c.from_index_space_id);
385  clog_assert(fitr != index_space_map_.end(), "invalid from index space");
386  const index_space_t & fi = fitr->second;
387 
388  auto titr = index_space_map_.find(c.to_index_space_id);
389  clog_assert(titr != index_space_map_.end(), "invalid to index space");
390  const index_space_t & ti = titr->second;
391 
392  auto citr = adjacency_map_.find(c.index_space_id);
393  clog_assert(citr == adjacency_map_.end(), "invalid adjacency info");
394  const adjacency_t & ci = citr->second;
395 
396  c.max_conn_size = fi.total_num_entities * ti.total_num_entities;
397 
398  // Create expanded index space
399  LegionRuntime::Arrays::Rect<2> expanded_bounds =
400  LegionRuntime::Arrays::Rect<2>(
401  LegionRuntime::Arrays::Point<2>::ZEROES(),
402  make_point(num_colors_, c.max_conn_size));
403 
404  Domain expanded_dom(Domain::from_rect<2>(expanded_bounds));
405  c.index_space = runtime_->create_index_space(ctx_, expanded_dom);
406  attach_name(c, c.index_space, "expanded index space");
407 
408  // Read user + FleCSI registered field spaces
409  c.field_space = runtime_->create_field_space(ctx_);
410 
411  FieldAllocator allocator =
412  runtime_->create_field_allocator(ctx_, c.field_space);
413 
414  for (const field_info_t & fi : context.registered_fields()) {
415  if (fi.index_space == c.index_space_id) {
416  allocator.allocate_field(fi.size, fi.fid);
417  }
418  }
419 
420  attach_name(c, c.field_space, "expanded field space");
421 
422  c.logical_region =
423  runtime_->create_logical_region(ctx_, c.index_space, c.field_space);
424  attach_name(c, c.logical_region, "expanded logical region");
425 
426  clog_assert(
427  adjacency_info.color_sizes.size() == num_colors_,
428  "mismatch in color sizes");
429 
430  DomainColoring color_partitioning;
431  for (size_t color = 0; color < num_colors_; ++color) {
432  LegionRuntime::Arrays::Rect<2> subrect(
433  make_point(color, 0),
434  make_point(color, adjacency_info.color_sizes[color] - 1));
435 
436  color_partitioning[color] = Domain::from_rect<2>(subrect);
437  }
438 
439  c.index_partition = runtime_->create_index_partition(
440  ctx_, c.index_space, color_domain_, color_partitioning,
441  true /*disjoint*/);
442  attach_name(c, c.index_partition, "color partitioning");
443 
444  adjacency_map_.emplace(adjacency_info.index_space, std::move(c));
445  }

509  void finalize(const indexed_coloring_info_map_t & indexed_coloring_info_map) {
510  using namespace std;
511 
512  using namespace Legion;
513  using namespace LegionRuntime;
514  using namespace Arrays;
515 
516  using namespace execution;
517 
518  context_t & context = context_t::instance();
519 
520  for (auto & itr : index_space_map_) {
521  index_space_t & is = itr.second;
522 
523  sparse_index_space_t* sis;
524 
525  auto sitr = sparse_index_space_map_.find(is.index_space_id);
526  if(sitr != sparse_index_space_map_.end()){
527  sis = &sitr->second;
528  }
529  else{
530  sis = nullptr;
531  }
532 
533  auto citr = indexed_coloring_info_map.find(is.index_space_id);
534  clog_assert(
535  citr != indexed_coloring_info_map.end(), "invalid index space");
536  const coloring_info_map_t & coloring_info_map = citr->second;
537 
538  FieldAllocator allocator =
539  runtime_->create_field_allocator(ctx_, is.field_space);
540 
541  auto ghost_owner_pos_fid = FieldID(internal_field::ghost_owner_pos);
542 
543  allocator.allocate_field(
544  sizeof(LegionRuntime::Arrays::Point<2>), ghost_owner_pos_fid);
545 
546  using field_info_t = context_t::field_info_t;
547 
548  for (const field_info_t & fi : context.registered_fields()) {
549  switch (fi.storage_class) {
550  case global:
551  case color:
552  case local:
553  case subspace:
554  break;
555  case ragged:
556  case sparse:
557  if (fi.index_space == is.index_space_id) {
558  if(utils::hash::is_internal(fi.key)){
559  allocator.allocate_field(fi.size, fi.fid);
560  }
561  else{
562  // this is a sparse offset
563  allocator.allocate_field(sizeof(utils::offset_t), fi.fid);
564  }
565  }
566  break;
567  default:
568  if (fi.index_space == is.index_space_id) {
569  allocator.allocate_field(fi.size, fi.fid);
570  }
571  break;
572  }
573  } // for
574 
575  is.logical_region =
576  runtime_->create_logical_region(ctx_, is.index_space, is.field_space);
577  attach_name(is, is.logical_region, "expanded logical region");
578 
579  // Partition expanded IndexSpace color-wise & create associated
580  // PhaseBarriers
581  DomainColoring color_partitioning;
582  for (int color = 0; color < num_colors_; color++) {
583  auto citr = coloring_info_map.find(color);
584  clog_assert(citr != coloring_info_map.end(), "invalid color info");
585  const coloring_info_t & color_info = citr->second;
586 
587  LegionRuntime::Arrays::Rect<2> subrect(
588  make_point(color, 0),
589  make_point(
590  color, color_info.exclusive + color_info.shared +
591  color_info.ghost - 1));
592 
593  color_partitioning[color] = Domain::from_rect<2>(subrect);
594  }
595 
596  is.index_partition = runtime_->create_index_partition(
597  ctx_, is.index_space, color_domain_, color_partitioning,
598  true /*disjoint*/);
599  attach_name(is, is.index_partition, "color partitioning");
600 
601  if(sis){
602  FieldAllocator allocator =
603  runtime_->create_field_allocator(ctx_, sis->field_space);
604 
605  for (const field_info_t & fi : context.registered_fields()) {
606  switch (fi.storage_class) {
607  case ragged:
608  case sparse:
609  if (fi.index_space == is.index_space_id) {
610  if(!utils::hash::is_internal(fi.key)){
611  allocator.allocate_field(fi.size + sizeof(size_t), fi.fid);
612  }
613  }
614  break;
615  default:
616  break;
617  }
618  } // for
619 
620  sis->logical_region = runtime_->create_logical_region(ctx_,
621  sis->index_space, sis->field_space);
622  attach_name(*sis, sis->logical_region, "sparse expanded logical region");
623 
624  DomainColoring color_partitioning;
625 
626  for (int color = 0; color < num_colors_; color++) {
627  auto citr = coloring_info_map.find(color);
628  clog_assert(citr != coloring_info_map.end(), "invalid color info");
629  const coloring_info_t & color_info = citr->second;
630 
631  LegionRuntime::Arrays::Rect<2> subrect(
632  make_point(color, 0), make_point(color, sis->color_size - 1));
633 
634  color_partitioning[color] = Domain::from_rect<2>(subrect);
635  }
636 
637  sis->index_partition = runtime_->create_index_partition(
638  ctx_, sis->index_space, color_domain_, color_partitioning,
639  true /*disjoint*/);
640 
641  attach_name(*sis, sis->index_partition, "sparse color partitioning");
642  }
643  }
644 
645  // create logical regions for color_index_space_
646  {
647  FieldAllocator allocator = runtime_->create_field_allocator(
648  ctx_, color_index_space_.field_space);
649 
650  using field_info_t = context_t::field_info_t;
651 
652  for (const field_info_t & fi : context.registered_fields()) {
653  if (fi.storage_class == color) {
654  allocator.allocate_field(fi.size, fi.fid);
655  } // if
656  } // for
657  color_index_space_.logical_region = runtime_->create_logical_region(
658  ctx_, color_index_space_.index_space, color_index_space_.field_space);
659  attach_name(
660  color_index_space_, color_index_space_.logical_region,
661  "color logical region");
662 
663  LegionRuntime::Arrays::Blockify<1> coloring(1);
664  color_index_space_.index_partition = runtime_->create_index_partition(
665  ctx_, color_index_space_.index_space, coloring);
666 
667  attach_name(
668  color_index_space_, color_index_space_.index_partition,
669  "color partitioning");
670  } // scope
671 
672  } // finalize
