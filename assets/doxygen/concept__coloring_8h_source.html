41 template<typename COLORING_POLICY>
42 void generic_coloring(
43  typename COLORING_POLICY::mesh_definition_t & md,
44  std::set<size_t> const & primary) {
45 
46  using namespace flecsi::topology;
47  using namespace flecsi::utils;
48 
49  int rank, size;
50  MPI_Comm_size(MPI_COMM_WORLD, &size);
51  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
52 
53  flecsi::coloring::index_coloring_t primary_coloring;
54  flecsi::coloring::coloring_info_t primary_coloring_info;
55 
56  constexpr size_t auxiliary_colorings = COLORING_POLICY::auxiliary_colorings;
57 
58  flecsi::coloring::index_coloring_t aux_coloring[auxiliary_colorings];
59  flecsi::coloring::coloring_info_t aux_coloring_info[auxiliary_colorings];
60 
61  constexpr size_t depth = COLORING_POLICY::primary::depth;
62  constexpr size_t dimension = COLORING_POLICY::primary::dimension;
63  constexpr size_t thru_dimension = COLORING_POLICY::primary::thru_dimension;
64 
65  // Create a communicator instance to get neighbor information.
66  auto communicator = std::make_shared<flecsi::coloring::mpi_communicator_t>();
67 
68  std::set<size_t> aggregate_near_neighbors;
69  auto core = primary;
70 
71  // Collect neighbors up to depth deep
72  for(size_t i{0}; i<depth; ++i) {
73 
74  // Form the closure of the current core
75  auto closure = entity_neighbors<dimension, dimension, thru_dimension>(
76  md, core);
77 
78  // Subtract off just the new nearest neighbors
79  auto nearest_neighbors = set_difference(closure, core);
80 
81  // Add these to the aggregate
82  aggregate_near_neighbors = set_union(aggregate_near_neighbors,
83  nearest_neighbors);
84 
85  // Update the core set
86  core = set_union(core, nearest_neighbors);
87  } // for
88 
89  // Form the closure of the collected neighbors
90  auto near_neighbor_closure =
91  entity_neighbors<dimension, dimension, thru_dimension>(
92  md, aggregate_near_neighbors);
93 
94  // Clean primaries from closure to get all neighbors
95  auto aggregate_neighbors =
96  set_difference(near_neighbor_closure, primary);
97 
98  if(rank == 0) {
99  std::cout << "aggregate near neighbors" << std::endl;
100  for(auto n: aggregate_near_neighbors) {
101  std::cout << n << std::endl;
102  } // for
103 
104  std::cout << "aggregate neighbors" << std::endl;
105  for(auto n: aggregate_neighbors) {
106  std::cout << n << std::endl;
107  } // for
108 
109  } // rank
110 
111  /*
112  Get the intersection of our near neighbors with the near neighbors
113  of other ranks. This map of sets will only be populated with
114  intersections that are non-empty
115  */
116  auto closure_intersection_map =
117  communicator->get_intersection_info(aggregate_near_neighbors);
118 
119  /*
120  Get the rank and offset information for our near neighbor
121  dependencies. This also gives information about the ranks
122  that access our shared entities.
123  */
124  auto primary_nn_info =
125  communicator->get_primary_info(primary, aggregate_near_neighbors);
126 
127  /*
128  Get the rank and offset information for all relevant neighbor
129  dependencies. This information will be necessary for determining
130  shared vertices.
131  */
132  auto primary_all_info =
133  communicator->get_primary_info(primary, aggregate_neighbors);
134 
135  // Create a map version of the local info for lookups below.
136  std::unordered_map<size_t, size_t> primary_indices_map;
137  {
138  size_t offset{0};
139  for(auto i: primary) {
140  primary_indices_map[offset++] = i;
141  } // for
142  } // scope
143 
144  // Create a map version of the remote info for lookups below.
145  std::unordered_map<size_t, flecsi::coloring::entity_info_t> remote_info_map;
146  for(auto i: std::get<1>(primary_all_info)) {
147  remote_info_map[i.id] = i;
148  } // for
149 
150  // Populate exclusive and shared primary information
151  {
152  size_t offset{0};
153 
154  for(auto i: std::get<0>(primary_nn_info)) {
155  if(i.size()) {
156  primary_coloring.shared.insert(
157  flecsi::coloring::entity_info_t(primary_indices_map[offset],
158  rank, offset, i));
159 
160  // Collect all colors with whom we require communication
161  // to send shared information.
162  primary_coloring_info.shared_users = flecsi::utils::set_union(
163  primary_coloring_info.shared_users, i);
164  }
165  else {
166  primary_coloring.exclusive.insert(
167  flecsi::coloring::entity_info_t(primary_indices_map[offset],
168  rank, offset, i));
169  } // if
170 
171  ++offset;
172  } // for
173  } // scope
174 
175  primary_coloring_info.exclusive = primary_coloring.exclusive.size();
176  primary_coloring_info.shared = primary_coloring.shared.size();
177  primary_coloring_info.ghost = primary_coloring.ghost.size();
178 
179  std::unordered_map<size_t, flecsi::coloring::entity_info_t>
180  shared_primary_map;
181  {
182  for(auto i: primary_coloring.shared) {
183  shared_primary_map[i.id] = i;
184  } // for
185  } // scope
186

188  // Lambda function to apply to auxiliary index spaces
189  //--------------------------------------------------------------------------//

191  auto color_entity = [&](size_t idx, auto tuple_element) {
192  using auxiliary_type = decltype(tuple_element);
193 
194  constexpr size_t index_space = auxiliary_type::index_space;
195  constexpr size_t dimension = auxiliary_type::dimension;
196  constexpr size_t primary_dimension = auxiliary_type::primary_dimension;
197 
198  std::cout << "idx: " << idx << std::endl;
199  std::cout << "index_space: " << index_space << std::endl;
200  std::cout << "dimension: " << dimension << std::endl;
201  std::cout << "primary_dimension: " << primary_dimension << std::endl;
202 
203  // Form the closure of this entity from the primary
204  auto auxiliary_closure =
205  entity_closure<primary_dimension, dimension>(md, near_neighbor_closure);
206 
207  using entity_info_t = flecsi::coloring::entity_info_t;
208 
209  // Assign entity ownership
210  std::vector<std::set<size_t>> entity_requests(size);
211  std::set<entity_info_t> entity_info;
212 
213  {
214  size_t offset{0};
215  for(auto i: auxiliary_closure) {
216  auto referencers =
217  entity_referencers<primary_dimension, dimension>(md, i);
218 
219  size_t min_rank(std::numeric_limits<size_t>::max());
220  std::set<size_t> shared_entities;
221 
222  // Iterate the direct referencers to assign entity ownership
223  for(auto c: referencers) {
224 
225  // Check the remote info map to see if this primary is
226  // off-color. If it is, compare it's rank for
227  // the ownership logic below.
228  if(remote_info_map.find(c) != remote_info_map.end()) {
229  min_rank = std::min(min_rank, remote_info_map.at(c).rank);
230  shared_entities.insert(remote_info_map.at(c).rank);
231  }
232  else {
233  // If the referencing primary isn't in the remote info map
234  // it is a local primary.
235 
236  // Add our rank to compare for ownership.
237  min_rank = std::min(min_rank, size_t(rank));
238 
239  // If the local primary is shared, we need to add all of
240  // the ranks that reference it.
241  if(shared_primary_map.find(c) != shared_primary_map.end())
242  shared_entities.insert(
243  shared_primary_map.at(c).shared.begin(),
244  shared_primary_map.at(c).shared.end()
245  );
246  } // if
247 
248  // Iterate through the closure intersection map to see if the
249  // indirect reference is part of another rank's closure, i.e.,
250  // that it is an indirect dependency.
251  for(auto ci: closure_intersection_map) {
252  if(ci.second.find(c) != ci.second.end()) {
253  shared_entities.insert(ci.first);
254  } // if
255  } // for
256 
257  if(min_rank == rank) {
258  // This is a entity that belongs to our rank.
259  auto entry = entity_info_t(i, rank, offset++, shared_entities);
260  entity_info.insert(entry);
261  }
262  else {
263  // Add remote entity to the request for offset information.
264  entity_requests[min_rank].insert(i);
265  } // if
266  } // for
267  } // for
268  } // scope
269 
270  auto entity_offset_info =
271  communicator->get_entity_info(entity_info, entity_requests);
272 
273  // Vertices index coloring.
274  for(auto i: entity_info) {
275  // if it belongs to other colors, its a shared entity
276  if(i.shared.size()) {
277  aux_coloring[idx].shared.insert(i);
278  // Collect all colors with whom we require communication
279  // to send shared information.
280  aux_coloring_info[idx].shared_users =
281  flecsi::utils::set_union(aux_coloring_info[idx].shared_users,
282  i.shared);
283  }
284  // otherwise, its exclusive
285  else
286  aux_coloring[idx].exclusive.insert(i);
287  } // for
288 
289  {
290  size_t r(0);
291  for(auto i: entity_requests) {
292 
293  auto offset(entity_offset_info[r].begin());
294  for(auto s: i) {
295  aux_coloring[idx].ghost.insert(entity_info_t(s, r, *offset));
296  // Collect all colors with whom we require communication
297  // to receive ghost information.
298  aux_coloring_info[idx].ghost_owners.insert(r);
299  // increment counter
300  ++offset;
301  } // for
302 
303  ++r;
304  } // for
305  } // scope
306 
307  }; // color_entity
308 
309  //--------------------------------------------------------------------------//
310  // End lambda
311  //--------------------------------------------------------------------------//
312 
313  tuple_visit<typename COLORING_POLICY::auxiliary>(color_entity);
314 
315 } // generic_coloring

354

356  auto colorer = std::make_shared<flecsi::coloring::parmetis_colorer_t>();
357 
358  // Generate the primary independent coloring
359  auto primary = colorer->color(dcrs);
360 
361  // Generate the rest of the colorings
362  generic_coloring<coloring_policy_t>(sd, primary);
