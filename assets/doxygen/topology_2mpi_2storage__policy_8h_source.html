94  void init_entities(
95  size_t domain,
96  size_t dim,
97  mesh_entity_base_ * entities,
98  utils::id_t * ids,
99  size_t size,
100  size_t num_entities,
101  size_t num_exclusive,
102  size_t num_shared,
103  size_t num_ghost,
104  bool read) {
105  auto & is = index_spaces[domain][dim];
106 
107  auto s = is.storage();
108  s->set_buffer(entities, num_entities, read);
109 
110  auto & id_storage = is.id_storage();
111  id_storage.set_buffer(ids, num_entities, true);
112 
113  for (auto & domain_connectivities : topology) {
114  auto & domain_connectivity__ = domain_connectivities[domain];
115  for (size_t d = 0; d <= NUM_DIMS; ++d) {
116  domain_connectivity__.get(d, dim).set_entity_storage(s);
117  } // for
118  } // for
119 
120  if (!read) {
121  return;
122  }
123 
124  is.set_end(num_entities);
125 
126  size_t shared_end = num_exclusive + num_shared;
127  size_t ghost_end = shared_end + num_ghost;
128 
129  for (size_t partition = 0; partition < num_partitions; ++partition) {
130  auto & isp = partition_index_spaces[partition][domain][dim];
131  isp.set_storage(s);
132  isp.set_id_storage(&id_storage);
133 
134  switch (partition_t(partition)) {
135  case exclusive:
136  isp.set_begin(0);
137  isp.set_end(num_exclusive);
138  break;
139  case shared:
140  isp.set_begin(num_exclusive);
141  isp.set_end(shared_end);
142  break;
143  case ghost:
144  isp.set_begin(shared_end);
145  isp.set_end(ghost_end);
146  break;
147  case owned:
148  isp.set_begin(0);
149  isp.set_end(shared_end);
150  break;
151  default:
152  break;
153  }
154  }
155  } // init_entities
