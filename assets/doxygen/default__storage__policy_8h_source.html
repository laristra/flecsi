312  void move( uintptr_t from, uintptr_t to ) {
313 
314  // check each namespace
315  for ( auto & sub_map : meta_ ) {
316 
317  // the namespace data
318  auto & namespace_key = sub_map.first;
319  auto & meta_data = sub_map.second;
320 
321  // create a temporary map
322  using map_type = typename std::decay< decltype( meta_data ) >::type;
323  map_type tmp_map;
324 
325  // loop over each element in the namespace and move
326  // matching ones into the temp map
327  auto itr = meta_data.begin();
328  while ( itr != meta_data.end() ) {
329  // get the meta data key and label
330  auto & meta_data_key = itr->first;
331  auto & label = itr->second.label;
332  // now build the hash for this label
333  auto key_hash = hash<const_string_t::hash_type_t>( label, label.size() );
334  auto from_hash = key_hash ^ from;
335  // test if it should be moved, and move it
336  if ( meta_data_key == from_hash ) {
337  auto to_hash = key_hash ^ to; // new hash
338  tmp_map[to_hash] = std::move( itr->second ); // move data
339  itr = meta_data.erase(itr); // delete old instance
340  }
341  // otherwise just go to next instance
342  else {
343  ++itr;
344  }
345  } // while
346 
347 
348  // get move iterators
349  using iterator = typename map_type::iterator;
350  using move_iterator = std::move_iterator<iterator>;
351 
352  // move the data back into the meta data map with the new key
353  meta_data.insert(
354  move_iterator( tmp_map.begin() ),
355  move_iterator( tmp_map.end() )
356  );
357 
358  } // for
