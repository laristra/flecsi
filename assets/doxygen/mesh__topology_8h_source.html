24 /*
25 
26  Description of major features and terms specific to FleCSI:
27 
28  mesh dimension MD - e.g: MD = 2 for a 2d mesh. We currently support 2d and 3d
29  meshes.
30 
31  topological dimension D - the dimensionality associated with entities,
32  e.g: D = 0 is interpreted as a vertex, D = 1 an edge or face for MD = 2,
33  D = 2 is a cell for MD = 2
34 
35  domain M - a sub-mesh or mesh space that holds entities of various topological
36  dimension
37 
38  connectivity - a directed connection or adjancy between entities of differing
39  topological dimension in the same domain. e.g: D1 -> D2 (edges -> faces)
40  for MD = 3. Cell to vertex connectivity is supplied by the user and all
41  other connectivies are computed by the topology.
42 
43  binding - a type of connectivity that connects entities of potentially
44  differing topological dimension across two different domains
45 
46  entity - an object associated with a topological dimension, e.g: cell. Each
47  entity has an associated integer id.
48 
49  mesh topology - the top-level container for domains, entities,
50  and connectivities, referred to as the low-level interface
51 
52  mesh policy - the top-level class that a specialization creates to
53  parameterize the mesh topology to define such things as: mesh dimension,
54  number of domains, connectivity and binding pairs of interest, and entity
55  classes/types per each domain/topological dimension.
56 
57  entity set - contains an iterable set of entities. Support set operations such
58  as intersection, union, etc. and functional operations like apply, map,
59  reduce, etc. to apply a custom function to the set.
60 
61 */

1140  template<typename I>
1141  void compute_graph_partition(
1142  size_t domain,
1143  size_t dim,
1144  const std::vector<I>& partition_sizes,
1145  std::vector<mesh_graph_partition<I>>& partitions){
1146 
1147  using int_t = I;
1148 
1149  partitions.reserve(partition_sizes.size());
1150 
1151  int_t total_size = 0;
1152  for(auto pi : partition_sizes){
1153  total_size += pi;
1154  }
1155 
1156  size_t n = num_entities_(dim, domain);
1157  size_t pn = n / total_size;
1158 
1159  size_t to_dim;
1160 
1161  if (dim == 0) {
1162  // vertex -> vertex via shared edge.
1163  to_dim = 1;
1164  } else {
1165  // edge -> edge via shared vertex, cell -> cell via shared edge/face etc.
1166  to_dim = dim - 1;
1167  }
1168 
1169  const connectivity_t& c1 = get_connectivity(domain, dim, to_dim);
1170  assert(!c1.empty() && "empty connectivity c1");
1171  const index_vector_t& fv1 = c1.get_from_index_vec();
1172 
1173  const connectivity_t& c2 = get_connectivity(domain, to_dim, dim);
1174  assert(!c2.empty() && "empty connectivity c2");
1175  const index_vector_t& fv2 = c2.get_from_index_vec();
1176 
1177  mesh_graph_partition<int_t> cp;
1178  cp.offset.reserve(pn);
1179 
1180  size_t offset = 0;
1181  size_t pi = 0;
1182 
1183  std::vector<int_t> partition;
1184  partition.push_back(0);
1185 
1186  for(size_t from_id = 0; from_id < n; ++from_id){
1187  auto to_ids = id_range(c1.get_entities(), fv1[from_id], fv1[from_id + 1]);
1188  cp.offset.push_back(offset);
1189 
1190  for(auto to_id : to_ids){
1191  auto ret_ids = id_range(c2.get_entities(), fv2[to_id.entity()], fv2[to_id.entity() + 1]);
1192 
1193  for(auto ret_id : ret_ids){
1194  if(ret_id.entity() != from_id){
1195  cp.index.push_back(ret_id.local_id());
1196  ++offset;
1197  }
1198  }
1199  }
1200 
1201  size_t m = cp.offset.size();
1202 
1203  if(m >= pn * partition_sizes[pi]){
1204  partitions.emplace_back(std::move(cp));
1205  partition.push_back(m + partition.back());
1206  offset = 0;
1207  ++pi;
1208  }
1209  }
1210 
1211  for(auto& pi : partitions){
1212  pi.partition = partition;
1213  }
1214  }
