370 template<typename T, typename U>
371 inline constexpr T
372 string_hash(U &&str, const std::size_t n)
373 {
374  if (n == 0)
375  return 0;
376 
377  // String-to-integer hash function, based on prime numbers.
378  // References:
379  // https://stackoverflow.com/questions/8317508/hash-function-for-a-string
380  // https://planetmath.org/goodhashtableprimes
381 
382  const T P = 3145739; // prime
383  const T Q = 6291469; // prime, a bit less than 2x the first
384 
385  T h = 37; // prime
386  for (std::size_t i = 0; i < n; ++i)
387  h = (h * P) ^ (str[i] * Q);
388  return h;
389 } // string_hash
