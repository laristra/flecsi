38  // A point is a node of an n-ary tree
39  template<size_t pt_start, size_t pt_end> struct point
40  {
41  static constexpr size_t start_ { pt_start };
42  static constexpr size_t end_ { pt_end };
43  static constexpr size_t count_ { end_ - start_ + 1 };
44  static constexpr bool is_end_point_ { count_ <= sequence_width };
45 
46  static constexpr size_t sequence_count()
47  {
48  return
49  points_in_sequence(sequence_width) > sequence_width
50  ?
51  sequence_width
52  :
53  points_in_sequence(sequence_width);
54  }
55 
56  private:
57  // Calculates the start and end indexes for a child node
58  static constexpr size_t child_start(size_t index)
59  {
60  return
61  index == 0
62  ?
63  pt_start
64  :
65  child_end(index - 1) + 1;
66  }
67  static constexpr size_t child_end(size_t index)
68  {
69  return
70  index == sequence_count() - 1
71  ?
72  pt_end
73  :
74  pt_start + points_in_sequence(sequence_count()) * (index + 1) -
75  (index < count_
76  ?
77  1
78  :
79  0);
80  }
81  static constexpr size_t points_in_sequence(size_t max)
82  {
83  return count_ / max + (
84  (count_ % max) > 0
85  ?
86  1
87  :
88  0);
89  }

116  // Loop function that calls the function passed to it, as well as recurses
117  template<size_t index>
118  static inline void flat_next
119  (std::integral_constant<size_t, index>, functor_types&&... functor_args)
120  {
121  flat_functor::template func<index>(std::forward<functor_types>(functor_args)...);
