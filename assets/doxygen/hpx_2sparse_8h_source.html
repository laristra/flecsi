382  void commit() {
383  if (!indices_) {
384  return;
385  } // if
386 
387  auto iitr = meta_data_.data.find(version_ | INDICES_FLAG);
388  assert(iitr != meta_data_.data.end());
389 
390  std::vector<uint8_t> & raw_indices =
391  const_cast<std::vector<uint8_t> &>(iitr->second);
392 
393  size_t * indices = reinterpret_cast<size_t *>(&raw_indices[0]);
394 
395  auto mitr = meta_data_.data.find(version_ | ENTRIES_FLAG);
396  assert(mitr != meta_data_.data.end());
397 
398  std::vector<uint8_t> & raw_entries =
399  const_cast<std::vector<uint8_t> &>(mitr->second);
400 
401  constexpr size_t ev_bytes = sizeof(entry_value_t);
402 
403  size_t s = raw_entries.size();
404  size_t c = raw_entries.capacity();
405  size_t d = (num_indices_ * num_slots_ + spare_map_.size()) * ev_bytes;
406 
407  if (c - s < d) {
408  // raw_entries.reserve(c * ev_bytes + d);
409  raw_entries.resize(c * ev_bytes + d);
410  } // if
411 
412  entry_value_t * entries =
413  reinterpret_cast<entry_value_t *>(&raw_entries[0]);
414 
415  entry_value_t * entries_end = entries + s / ev_bytes;
416 
417  for (size_t i = 0; i < num_indices_; ++i) {
418  size_t n = indices_[i];
419 
420  size_t pos = indices[i];
421 
422  if (n == 0) {
423  indices[i + 1] = pos;
424  continue;
425  } // if
426 
427  auto p = spare_map_.equal_range(i);
428 
429  size_t m = distance(p.first, p.second);
430 
431  entry_value_t * start = entries_ + i * num_slots_;
432  entry_value_t * end = start + n;
433 
434  auto iitr = entries + pos;
435 
436  std::copy(iitr, entries_end, iitr + n + m);
437  std::copy(start, end, iitr);
438  entries_end += n + m;
439 
440  auto cmp = [](const auto & k1, const auto & k2) -> bool {
441  return k1.entry < k2.entry;
442  };
443 
444  auto nitr = iitr + indices[i + 1];
445 
446  std::inplace_merge(iitr, nitr, nitr + n, cmp);
447 
448  if (m == 0) {
449  indices[i + 1] = pos + n;
450  continue;
451  } // if
452 
453  indices[i + 1] = pos + n + m;
454 
455  auto vitr = nitr + n;
456 
457  for (auto itr = p.first; itr != p.second; ++itr) {
458  vitr->entry = itr->second.entry;
459  vitr->value = itr->second.value;
460  ++vitr;
461  } // for
462 
463  std::inplace_merge(iitr, nitr + n, nitr + n + m, cmp);
464  } // for
465 
466  delete[] indices_;
467  indices_ = nullptr;
468 
469  delete[] entries_;
470  entries_ = nullptr;
471 
472  if (!erase_set_) {
473  return;
474  }
475 
476  constexpr size_t erased_marker = std::numeric_limits<size_t>::max();
477 
478  entry_value_t * start;
479  entry_value_t * end;
480 
481  size_t last = std::numeric_limits<size_t>::max();
482 
483  size_t num_erased = 0;
484 
485  for (auto p : *erase_set_) {
486  if (p.first != last) {
487  start = entries + indices[p.first];
488  end = entries + indices[p.first + 1];
489  last = p.first;
490  indices[p.first] -= num_erased;
491  }
492 
493  entry_value_t * m = std::lower_bound(
494  start, end, entry_value_t(p.second),
495  [](const auto & k1, const auto & k2) -> bool {
496  return k1.entry < k2.entry;
497  });
498 
499  m->entry = erased_marker;
500  start = m + 1;
501  ++num_erased;
502  }
503 
504  while (last < num_indices_) {
505  indices[++last] -= num_erased;
506  }
507 
508  std::remove_if(entries, entries_end, [](const auto & k) -> bool {
509  return k.entry == erased_marker;
510  });
511 
512  s = raw_entries.size();
513  s -= num_erased * ev_bytes;
514  raw_entries.resize(s);
515 
516  delete erase_set_;
517  erase_set_ = nullptr;
518  } // commit
