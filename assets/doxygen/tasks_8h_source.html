33 
34  auto & vertex_map { context.index_map(index_spaces::vertices) };
35  auto & reverse_vertex_map
36  { context.reverse_index_map(index_spaces::vertices) };
37  auto & cell_map { context.index_map(index_spaces::cells) };
38 
39  std::vector<vertex_t *> vertices;
40 
41  const size_t width { input_mesh_dimension_x };
42  const double dt { 1.0/width };
43 
44  for(auto & vm: vertex_map) {
45  const size_t mid { vm.second };
46  const size_t row { mid/(width+1) };
47  const size_t column { mid%(width+1) };
48 
49  point_t p { column*dt, row*dt };
50  vertices.push_back(m.make<vertex_t>(p));
51  } // for
52 
53  size_t count{0};
54  for(auto & cm: cell_map) {
55  const size_t mid { cm.second };
56 
57  const size_t row { mid/width };
58  const size_t column { mid%width };
59 
60  const size_t v0 { (column ) + (row ) * (width + 1) };
61  const size_t v1 { (column + 1) + (row ) * (width + 1) };
62  const size_t v2 { (column + 1) + (row + 1) * (width + 1) };
63  const size_t v3 { (column ) + (row + 1) * (width + 1) };
64 
65  const size_t lv0 { reverse_vertex_map[v0] };
66  const size_t lv1 { reverse_vertex_map[v1] };
67  const size_t lv2 { reverse_vertex_map[v2] };
68  const size_t lv3 { reverse_vertex_map[v3] };
69 
70  auto c { m.make<cell_t>() };
71  m.init_cell<0>(c, { vertices[lv0], vertices[lv1],
72  vertices[lv2], vertices[lv3] });
73  } // for
74 
75  m.init<0>();
76 } // initizlize_mesh
