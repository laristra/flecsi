43 entity_neighbors(const mesh_definition_u<D> & md, size_t entity_id) {
44  // Get the vertices of the requested id
45  auto vertices = md.entities_set(from_dim, 0, entity_id);
46 
47  // Put the results into set form
48  std::set<size_t> neighbors;
49 
50  // Go through the entities of the to_dim
51  for (size_t e(0); e < md.num_entities(to_dim); ++e) {
52

54  if (from_dim == to_dim && e == entity_id) {
55  continue;
56  } // if
57

97 std::set<size_t>
98 entity_neighbors(const mesh_definition_u<D> & md, U && indices) {
99  clog_assert(from_dim == to_dim, "from_dim does not equal to to_dim");
100 
101  // Closure should include the initial set
102  std::set<size_t> closure(
103  std::forward<U>(indices).begin(), std::forward<U>(indices).end());
104 
105  using entityid = size_t;
106  using vertexid = size_t;
107 
108  // essentially vertex to cell and cell to cell through vertices
109  // connectivity.
110  std::map<vertexid, std::vector<entityid>> vertex2entities;
111  std::map<entityid, std::vector<entityid>> entity_neighbors;
112 
113  // build global entity to entity through # of shared vertices connectivity
114  for (size_t cell(0); cell < md.num_entities(from_dim); ++cell) {
115  std::map<entityid, size_t> cell_counts;
116  for (auto vertex : md.entities(from_dim, 0, cell)) {
117  // build vertex to cell connectivity, O(n_cells * n_polygon_sides)
118  // of insert().
119  vertex2entities[vertex].push_back(cell);
120 
121  // Count the number of times this cell shares a common vertex with
122  // some other cell. O(n_cells * n_polygon_sides * vertex to cells degrees)
123  for (auto other : vertex2entities[vertex]) {
124  // for (auto other : md.entities(0, from_dim, vertex)) {
125  if (other != cell)
126  cell_counts[other] += 1;
127  }
128  }
129 
130  for (auto count : cell_counts) {
131  if (count.second > thru_dim) {
132  // append cell to cell through "dimension" connectivity, we need to
133  // add both directions
134  entity_neighbors[cell].push_back(count.first);
135  entity_neighbors[count.first].push_back(cell);
136  }
137  }
138  }
139 
140  for (auto i : indices) {
141  for (auto n : entity_neighbors[i]) {
142  closure.insert(n);
143  }
144  }
145 
146  return closure;
147 } // entity_closure
