105 template<
106  std::size_t DIMENSION,
107  std::size_t FROM_DIMENSION = DIMENSION,
108  std::size_t TO_DIMENSION = DIMENSION,
109  std::size_t THRU_DIMENSION = DIMENSION - 1>
110 inline dcrs_t
111 make_dcrs(const typename topology::mesh_definition_u<DIMENSION> & md) {
112  int size;
113  int rank;
114 
115  MPI_Comm_size(MPI_COMM_WORLD, &size);
116  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
117 
118  //--------------------------------------------------------------------------//
119  // Create a naive initial distribution of the indices
120  //--------------------------------------------------------------------------//
121 
122  size_t quot = md.num_entities(FROM_DIMENSION) / size;
123  size_t rem = md.num_entities(FROM_DIMENSION) % size;
124 
125  // Each rank gets the average number of indices, with higher ranks
126  // getting an additional index for non-zero remainders.
127  size_t init_indices = quot + ((rank >= (size - rem)) ? 1 : 0);
128 
129  // Start to initialize the return object.
130  dcrs_t dcrs;
131  dcrs.distribution.push_back(0);
132 
133  // Set the distributions for each rank. This happens on all ranks.
134  for (size_t r(0); r < size; ++r) {
135  const size_t indices = quot + ((r >= (size - rem)) ? 1 : 0);
136  dcrs.distribution.push_back(dcrs.distribution[r] + indices);
137  } // for
138 
139  //--------------------------------------------------------------------------//
140  // Create the cell-to-cell graph.
141  //--------------------------------------------------------------------------//
142 
143  // Set the first offset (always zero).
144  dcrs.offsets.push_back(0);
145 
146  // Add the graph adjacencies by getting the neighbors of each
147  // cell index
148  using cellid = size_t;
149  using vertexid = size_t;
150 
151  // essentially vertex to cell and cell to cell through vertices
152  // connectivity.
153  std::map<vertexid, std::vector<cellid>> vertex2cells;
154  std::map<cellid, std::vector<cellid>> cell2cells;
155 
156  // build global cell to cell through # of shared vertices connectivity
157  for (size_t cell(0); cell < md.num_entities(FROM_DIMENSION); ++cell) {
158  std::map<cellid, size_t> cell_counts;
159 
160  for (auto vertex : md.entities(FROM_DIMENSION, 0, cell)) {
161  // build vertex to cell connectivity, O(n_cells * n_polygon_sides)
162  // of insert().
163  vertex2cells[vertex].push_back(cell);
164 
165  // Count the number of times this cell shares a common vertex with
166  // some other cell. O(n_cells * n_polygon_sides * vertex to cells degrees)
167  for (auto other : vertex2cells[vertex]) {
168  if (other != cell)
169  cell_counts[other] += 1;
170  }
171  }
172 
173  for (auto count : cell_counts) {
174  if (count.second > THRU_DIMENSION) {
175  // append cell to cell through "dimension" connectivity, we need to
176  // add both directions
177  cell2cells[cell].push_back(count.first);
178  cell2cells[count.first].push_back(cell);
179  }
180  }
181  }
182 
183  // turn subset of cell 2 cell connectivity to dcrs
184  for (size_t i(0); i < init_indices; ++i) {
185  auto cell = dcrs.distribution[rank] + i;
186 
187  for (auto n : cell2cells[cell]) {
188  dcrs.indices.push_back(n);
189  } // for
190 
191  dcrs.offsets.push_back(dcrs.offsets[i] + cell2cells[cell].size());
192  }
193 
194  return dcrs;
195 } // make_dcrs
